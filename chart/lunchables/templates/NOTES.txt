To run administrative commands, save and use the following shell function:

function pipeline-adm () {
    if [ -t 0 -a -t 1 ]; then
        EXTRA_ARGS="-t"
    else
        EXTRA_ARGS=""
    fi
    kubectl exec -i $EXTRA_ARGS -n {{ .Release.Namespace }} deploy/{{ .Release.Name }}-leader-adm -- python3 leader.py "$@"
}
